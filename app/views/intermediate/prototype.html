Prototype

A object always automatically inherits the properties and methods of it's parents, never from a child.

Object Base

{
  honk = function() {
    console.log('Beep beep!');
  }
};

Car

function Car(make, model) {
  this.make = make;
  this.model = model;
  this.accelerate = function() {
    console.log('This ' + this.model + ' is accelerating to 60mph.');
  }
  const driver = 'Michael Knight';
};

//Creating an instance of a Car
const kitt = new Car('Pontiac','Trans Am');

//Because objects inherit methods and properties from it's parent and never from it's child, this is why you can call kitt.accelerate() AND kitt.honk().
kitt.accelerate();
  =>This Trans Am is accelerating to 60mph.
kitt.honk();
  => Beep beep!

//But you may NOT call kitt.__proto__.__proto__.accelerate()
kitt.__proto__.__proto__.accelerate()
  => prototype.js:21 Uncaught TypeError: kitt.__proto__.__proto__.accelerate is not a function
